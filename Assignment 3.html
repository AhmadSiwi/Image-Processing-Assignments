def harrisCornerDetector(img, windowsize = 5, k = 0.05, thresh = 0.2):
    #Compute Sobel derivative
    dx = cv2.Sobel(img,cv2.CV_64F,1,0,ksize=windowsize)
    dy = cv2.Sobel(img,cv2.CV_64F,0,1,ksize=windowsize)
    
    height = img.shape[0]
    width = img.shape[1]
    #Compute Product of derivatives
    Ixx = dx ** 2
    Iyy = dy ** 2
    Ixy = dx * dy
    
    offset = int(windowsize / 2)
    
    RImage = np.zeros(img.shape)
    corners = []
    for i in range(offset, height - offset):
        for j in range(offset, width - offset):
            windowIxx = Ixx[i - offset:i + offset + 1, j - offset:j + offset + 1]
            windowIyy = Iyy[i - offset:i + offset + 1, j - offset:j + offset + 1]
            windowIxy = Ixy[i - offset:i + offset + 1, j - offset:j + offset + 1]
            
            Sxx = windowIxx.sum()
            Syy = windowIyy.sum()
            Sxy = windowIxy.sum()
        
            det = Sxx * Syy - Sxy * Sxy
            trace = Sxx + Syy
            
            R = det - k * (trace ** 2)
            
            RImage[i, j] = R
    
    #normalize R values
    RImage = RImage / np.max(RImage)
    
    for i in range(offset, height - offset):
        for j in range(offset, width - offset):
            R = RImage[i, j]
            if R > thresh:
                corners.append((R, j, i))
                
    return corners

def MyHoughLines(edges,threshold):
    H = {}
    (x, y) = edges.shape
    for i in range(x):
        for j in range(y):
            if edges[i][j] == 255:
                for theta in range(181):
                    theta_d = np.pi*theta/180
                    d =  i * np.cos(theta_d) + j * np.sin(theta_d)
                    if((d,theta_d) in H.keys()):
                        H[(d,theta_d)] += 1
                    else:
                        H[(d,theta_d)] = 1
    lines = []
    for k, v in H.items():
        if v>= threshold:
            lines.append(k)
    return lines

def MyHoughCircles(image):
    edges = cv2.Canny(image,50,150,apertureSize = 3)
    H = {}
    (x, y) = edges.shape
    r_max = int(min(x,y)/2)
    #For every edge pixel (i,j)
    for i in range(1,x-1):
        for j in range(1,y-1):
            if edges[i][j]==0:
                continue
            #estimated gradient at (x,y)
            dx = edges[i+1][j-1] + 2*edges[i+1][j] + edges[i+1][j+1] - edges[i-1][j-1] - 2*edges[i-1][j] - edges[i-1][j+1]
            dy = edges[i-1][j+1] + 2*edges[i][j+1] + edges[i+1][j+1] - edges[i-1][j-1] - 2*edges[i][j-1] - edges[i+1][j-1]
            # +0.0001 to avoid dividing by zero
            theta = np.arctan(dy/(dx+0.0001))
            #For each possible radius value r
            for r in range(-1*r_max,r_max):
                a = int(i + r*np.cos(theta))
                b = int(j + r*np.sin(theta))
                if (a<0 or b<0 or a>=x or b>=y):
                    continue
                if ((a,b,r) in H.keys()):
                    H[(a,b,r)] += 1
                else:
                    H[(a,b,r)] = 1
    circles = []
    for k, v in H.items():
        if v >= 10:
            circles.append(k)
    return circles
