def Inkwell(image):
    i = image.copy()
    return cv2.cvtColor(i, cv2.COLOR_BGR2GRAY)

def Lily(image):
    i = image.copy()
    i[:,:,1] = np.maximum(127,i[:,:,1])
    return i

def Sutro(image):
    i = image.copy()
    j = i .copy()
    i[:,:,0] = np.minimum(j[:,:,0],j[:,:,1],j[:,:,2])
    i[:,:,1] = np.minimum(j[:,:,0],j[:,:,1],j[:,:,2])
    return i

def Gotham(image):
    i = image.copy()
    j = Inkwell(i)
    j = j-128
    j = np.multiply(j,(j<=127))
    j = j*2
    #To make dark pixels black and reserve the white ones as white
    return j

def insta_like(image, instafilter):
    if (instafilter == 'Lily'):
        image = Lily(image)
    if (instafilter == 'Inkwell'):
        image = Inkwell(image)
    if (instafilter == 'Original'):
        image = Original(image)
    if (instafilter == 'Sutro'):
        image = Sutro(image)
    if (instafilter == 'Gotham'):
        image = Gotham(image)
    return image
